Simple implementation problem. Find the number of digits from the front of `s` that match digits from pi.
<CH.Code>

```cpp 
#include <iostream>
#include <string>

constexpr std::string_view kDigitsOfPi = "314159265358979323846264338327";

void solve() {
  std::string s;
  std::cin >> s;
  for (int i = 0; i < kDigitsOfPi.length(); ++i) {
    if (s[i]!=kDigitsOfPi[i]) {
      std::cout << i << '\n';
      return;
    }
  }
  std::cout << "30\n";
}

int main() {
  int t;
  std::cin >> t;
  for (int tc = 1; tc <= t; ++tc) {
    solve();
  }
  return 0;
}

```
